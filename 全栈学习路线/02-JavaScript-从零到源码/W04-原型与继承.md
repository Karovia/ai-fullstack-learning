# W04 - 原型与继承

> 本周目标：把 JS 那个被骂了 30 年的"原型链"看明白。结束时你能手写 `myNew / myInstanceof / Object.create`，能用 ES5 原型实现一套 ES6 class 等价物。

## 1. 一句话讲原型

**每个 JS 对象都有一个隐藏指针 `__proto__`，指向"父母对象"。访问属性找不到时，沿 `__proto__` 一路往上找，直到 `null`。这条链就是原型链。**

> **类比 Python**：Python 类的 `MRO`（方法解析顺序）也是从子类找到父类。但 Python 是"类找父类"，JS 是"对象找对象"——更动态、更灵活、也更怪。

```javascript
const animal = { eat() { console.log('吃东西'); } };
const dog = { bark() { console.log('汪'); } };
Object.setPrototypeOf(dog, animal);     // dog.__proto__ = animal

dog.bark();   // 汪      自己有
dog.eat();    // 吃东西   原型链找到 animal
dog.fly;      // undefined  没有，链尽头返回 undefined
```

## 2. `__proto__` vs `prototype`（图解）

这是劝退新手的最大障碍：

| 名字 | 谁有 | 作用 |
|------|------|------|
| `__proto__`（或 `[[Prototype]]`） | 每个**对象**都有 | 指向自己的"父母" |
| `prototype` | 只有**函数**才有 | 用 `new` 时给新对象当父母 |

```javascript
function Dog(name) { this.name = name; }
Dog.prototype.bark = function() { console.log(this.name + ' 汪'); };

const d = new Dog('旺财');
d.bark();                              // 旺财 汪
d.__proto__ === Dog.prototype;         // true  ✅
Dog.prototype.__proto__ === Object.prototype;  // true
Object.prototype.__proto__ === null;   // true  链终点
```

图解：

```
d  ──__proto__──▶  Dog.prototype  ──__proto__──▶  Object.prototype  ──▶  null
```

## 3. `new` 操作符做了什么（4 步）

```javascript
function Dog(name) { this.name = name; }
const d = new Dog('旺财');
```

`new` 内部等价于：
1. 创建一个空对象 `obj = {}`
2. 把 `obj.__proto__` 指向 `Dog.prototype`
3. 用 `obj` 当 `this` 调用 `Dog('旺财')`
4. 如果构造函数返回的是对象就用那个，否则用 `obj`

### 手写 `myNew`

```javascript
function myNew(Ctor, ...args) {
  const obj = Object.create(Ctor.prototype);   // 步骤 1+2
  const result = Ctor.apply(obj, args);         // 步骤 3
  return (result && typeof result === 'object') ? result : obj;  // 步骤 4
}

function Dog(name) { this.name = name; }
Dog.prototype.bark = function() { return this.name + ' 汪'; };

const d = myNew(Dog, '旺财');
d.bark();   // '旺财 汪'
```

## 4. `instanceof` 原理

`a instanceof B`：沿着 `a.__proto__` 的链找，看是否能找到 `B.prototype`。

### 手写 `myInstanceof`

```javascript
function myInstanceof(obj, Ctor) {
  let proto = Object.getPrototypeOf(obj);
  while (proto !== null) {
    if (proto === Ctor.prototype) return true;
    proto = Object.getPrototypeOf(proto);
  }
  return false;
}

myInstanceof([], Array);    // true
myInstanceof([], Object);   // true（Array 继承 Object）
myInstanceof([], Function); // false
```

## 5. `Object.create` 原理

```javascript
// 等价实现
function myCreate(proto) {
  function F() {}
  F.prototype = proto;
  return new F();
}
const animal = { eat() {} };
const dog = myCreate(animal);
dog.__proto__ === animal;   // true
```

## 6. ES5 六种继承（历史课，了解即可）

```javascript
function Animal(name) { this.name = name; this.colors = ['red']; }
Animal.prototype.eat = function() { console.log(this.name + ' 吃'); };

// ❌ 1. 原型链继承：共享引用 + 不能传参
function Dog1() {}
Dog1.prototype = new Animal();

// ❌ 2. 借用构造：能传参，但拿不到原型方法
function Dog2(name) { Animal.call(this, name); }

// ⚠️ 3. 组合继承：上面两种合体，但调用 Animal 两次
function Dog3(name) { Animal.call(this, name); }
Dog3.prototype = new Animal();
Dog3.prototype.constructor = Dog3;

// 4. 原型式（≈ Object.create）
// 5. 寄生式：在原型式上加方法
// 6. 寄生组合（最优解，ES6 class 的本质）
function Dog6(name) { Animal.call(this, name); }
Dog6.prototype = Object.create(Animal.prototype);
Dog6.prototype.constructor = Dog6;
```

> 真实工作中**直接用 ES6 class**，这些只为面试和读老代码做准备。

## 7. ES6 class 是语法糖

```javascript
class Animal {
  constructor(name) { this.name = name; }
  eat() { console.log(this.name + ' 吃'); }
  static species() { return '动物'; }       // 静态方法（挂在类上）
}

class Dog extends Animal {
  #secret = '骨头藏在花园';                  // 私有字段（# 开头）
  constructor(name, breed) {
    super(name);                             // 必须先调 super
    this.breed = breed;
  }
  bark() { console.log(this.name + ' 汪'); }
  get info() { return `${this.name}(${this.breed})`; }  // getter
  set info(v) { [this.name, this.breed] = v.split('-'); }
}

const d = new Dog('旺财', '柴犬');
d.eat();             // 旺财 吃    继承
d.bark();            // 旺财 汪
d.info;              // 旺财(柴犬)  getter
d.info = '小黑-黑贝';
Animal.species();    // 动物
// d.#secret;        // ❌ 类外部不能访问
```

`extends` 等价于：
```javascript
Dog.prototype = Object.create(Animal.prototype);
Dog.prototype.constructor = Dog;
Object.setPrototypeOf(Dog, Animal);   // 静态方法继承
```

## 8. `super` 的两种用法

```javascript
class Cat extends Animal {
  constructor(name) {
    super(name);                  // 1. 当函数调：调父类构造器
  }
  eat() {
    super.eat();                  // 2. 当对象调：访问父类方法
    console.log('+ 喝水');
  }
}
```

## 9. 静态方法 / 实例方法 / 私有字段

```javascript
class Counter {
  static count = 0;            // 静态字段
  #value;                      // 私有字段
  
  constructor() {
    Counter.count++;
    this.#value = 0;
  }
  
  inc() { return ++this.#value; }   // 实例方法
  
  static reset() { Counter.count = 0; }  // 静态方法
}
```

## 10. 实战练习

### 练习 1：手写完整的 `myNew`（已在第 3 节）

### 练习 2：模拟 `class extends`（用 ES5 实现）

```javascript
function inherit(Sub, Super) {
  Sub.prototype = Object.create(Super.prototype);
  Sub.prototype.constructor = Sub;
  Object.setPrototypeOf(Sub, Super);
}

function Animal(name) { this.name = name; }
Animal.prototype.eat = function() { return this.name + ' 吃'; };

function Dog(name, breed) {
  Animal.call(this, name);
  this.breed = breed;
}
inherit(Dog, Animal);
Dog.prototype.bark = function() { return this.name + ' 汪'; };

const d = new Dog('旺财', '柴犬');
d.eat();   // '旺财 吃'
d.bark();  // '旺财 汪'
d instanceof Dog;     // true
d instanceof Animal;  // true
```

### 练习 3：实现一个简单的 EventTarget 体系

```javascript
class Animal {
  constructor(name) { this.name = name; this.listeners = {}; }
  on(evt, fn) { (this.listeners[evt] ??= []).push(fn); return this; }
  emit(evt, ...args) {
    (this.listeners[evt] || []).forEach(fn => fn.call(this, ...args));
    return this;
  }
}

class Dog extends Animal {
  bark() { this.emit('bark', this.name); return this; }
}

const d = new Dog('旺财');
d.on('bark', name => console.log(`${name} 在叫！`)).bark();
// 旺财 在叫！
```

## 常见错误

1. **混淆 `__proto__` 和 `prototype`**：对象有前者，函数才有后者
2. **箭头函数当方法挂在 prototype**：箭头无 this，会出错
3. **忘记 `super(...)`**：子类构造器必须先调，否则 `this` 报错
4. **修改原型对象会影响所有实例**：如 `Array.prototype.foo = 1` → 全局污染
